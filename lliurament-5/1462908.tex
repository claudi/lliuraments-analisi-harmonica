\begin{enumerate}
    \item[\textbf{1.}] Suposeu que~\(\psi\in\S(\RR^{d})\)
        satisfà~\(\int\abs{\psi(x)}^{2}=\diff x=1\). Demostreu que
        \[
            \biggl(\int_{\RR^{d}}\abs{x}^{2}\abs{\psi(x)}^{2}\diff x\biggr)
            \biggl(\int_{\RR^{d}}\abs{\xi}^{2}\abs{\widehat{\psi}(\xi)}^{2}\diff\xi\biggr)
            \geq
            \frac{d^{2}}{16\uppi^{2}}.
        \]
\end{enumerate}
